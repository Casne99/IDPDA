\begin{frame}{Input Driven Pushdown Automata}
Un Input Driven Pushdown Automata deterministico è una settupla
\[
A=(\Sigma,Q,\Gamma,q_0,\perp,[\delta_a]_{a\in\Sigma},F),\quad
\Sigma = \Sigma_{+1} \cup \Sigma_{-1} \cup \Sigma_0.
\]

\vfill
\begin{columns}[T,onlytextwidth]
  \column{0.5\textwidth}
  \begin{itemize}
    \item $\Sigma_{+1}$: parentesi sinistre
    \item $\Sigma_{-1}$: parentesi destre
    \item $\Sigma_0$: simboli neutri
  \end{itemize}

  \column{0.5\textwidth}
  \begin{itemize}
    \item $\delta_{<}:Q\to Q\times\Gamma$.
    \item $\delta_{>}:Q\times(\Gamma\cup\{\perp\})\to Q$.
    \item $\delta_c:Q\to Q$.
  \end{itemize}
\end{columns}

\vfill

    A. Okhotin, K. Salomaa, 2014.
\end{thebibliography}
\end{frame}

\begin{frame}{Distanza stringa-linguaggio e correction problem}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Idea:} dati $x \in \Sigma^*$ e $L \subseteq \Sigma^*$, trovare la stringa $y \in L$ da cui $x$ ha distanza minima.

            \vspace{1em}
            \[
                d(L,x)=\min_{w\in L} d(x,w).
            \]

            \vspace{1em}
            \textbf{Correction problem:} trovare una correzione ottima
            \[
                y=\arg\min_{w\in L} d(x,w).
            \]
        \end{column}
\begin{column}{0.5\textwidth}
            \centering
            \begin{tikzpicture}[scale=0.9, transform shape]
                % Stringhe in L sparpagliate
                \node at (0, 1.2) {UNIMI};

            \end{tikzpicture}

            \]
        \end{column}

\begin{frame}{$k$-vicinato di un linguaggio $L$}
Definizione induttiva del vicinato di edit:
\[
E_0(w)=\{w\},\quad
E_{k+1}(w)=E_k(w)\cup\bigcup_{w'\in E_k(w)}\bigcup_{a\in\Sigma}(ins_a(w')\cup del_a(w')).
\]

Per un linguaggio:
\[
E_k(L)=\bigcup_{w\in L}E_k(w).
\]

\begin{itemize}
  \item Risultato: la famiglia dei linguaggi IDPDA e' chiusa rispetto al $k$-vicinato.
\end{itemize}

\begin{frame}{Linguaggi con distanza di edit "facile"}
    \begin{itemize}
        \setlength{\itemsep}{1em}
        \item $O(n)$ per linguaggi regolari.
        \item $O(n^3)$ per linguaggi context-free.
        \begin{itemize}
            \item[$\rightarrow$] Caso particolare: linguaggi accettati da Input Driven Pushdown Automata.
        \end{itemize}
        \item La classe dei linguaggi riconosciuti da 1-NAuxPDA è la più ampia classe per cui si conosce un algoritmo efficiente per il calcolo della distanza di edit.
    \end{itemize}

\begin{frame}{Linguaggi con distanza di edit "difficile"}
\begin{columns}[T,onlytextwidth]
    \column{0.5\textwidth}
    \begin{itemize}
        \item P: problemi risolvibili in tempo polinomiale deterministico.
        \item NP: problemi con certificati verificabili in tempo polinomiale deterministico.
    \end{itemize}
    \column{0.5\textwidth}
    \vfill
    \centering
    Questione aperta: $\stackrel{?}{P=NP}$
    \vfill
\end{columns}

\begin{block}{Teorema}
Se $P=NP$, allora la distanza di edit di qualunque linguaggio in $NP$ è calcolabile in tempo polinomiale.
\end{block}

Idea della prova:
\begin{itemize}
  \item Si considera il problema decisionale $\Pi_L$: ``$d(L,x)\le k$?''.
  \item Si indovina una stringa candidata $y$ di lunghezza polinomiale in $|x|$.
  \item Si verifica se $y \in L$ e $d(y, x) \le k$.
  \item Si calcola $d(L,x)$ valutando $\Pi_L$ per valori crescenti di $k$.
\end{itemize}
\end{frame}
