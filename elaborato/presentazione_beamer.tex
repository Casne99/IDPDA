\begin{frame}{Obiettivi della tesi}
\begin{itemize}
  \item Studiare il modello degli \emph{Input Driven Pushdown Automata} (IDPDA).
  \item Studiare il problema del calcolo della distanza di edit stringa-stringa e stringa-linguaggio.
  \item Evidenziare collegamenti tra automi input-driven e problema del calcolo delle distanze di edit.
  \item Evidenziare legami tra distanze di edit e questioni aperte in teoria della complessità.
\end{itemize}

\begin{frame}{DIDPDA: alfabeto e definizione}
Un DIDPDA e' una settupla
\[
A=(\Sigma,Q,\Gamma,q_0,\perp,[\delta_a]_{a\in\Sigma},F),\quad
\Sigma = \Sigma_{+1} \cup \Sigma_{-1} \cup \Sigma_0.
\]

\vfill
\begin{columns}[T,onlytextwidth]
  \column{0.5\textwidth}
  \begin{itemize}
    \item $\Sigma_{+1}$: parentesi sinistre
    \item $\Sigma_{-1}$: parentesi destre
    \item $\Sigma_0$: simboli neutri
  \end{itemize}

  \column{0.5\textwidth}
  \begin{itemize}
    \item $\delta_{<}:Q\to Q\times\Gamma$.
    \item $\delta_{>}:Q\times(\Gamma\cup\{\perp\})\to Q$.
    \item $\delta_c:Q\to Q$.
  \end{itemize}
\end{columns}
\end{frame}

\begin{frame}{Distanze di edit: definizioni}

  \begin{itemize}
    \setlength{\itemsep}{0.55em}
    \item \textbf{Operazione di edit:} coppia $(a,b)\neq(\epsilon,\epsilon)$ con $|a|,|b|\le 1$, indicata con $a\rightarrow b$.
    \item \textbf{Sequenza di edit:} $S=(s_1,\dots,s_k)$ con costo $\mathrm{cost}(S)=\sum_{i=1}^{k}\gamma(s_i)$.
    \item $y$ si ottiene da $x$ con $a\rightarrow b$ se $\exists v,w\in\Sigma^*$ tali che $x=vaw$ e $y=vbw$.
  \end{itemize}

  \[
    d(x,y)=\min\{\mathrm{cost}(S)\mid S\text{ trasforma }x\text{ in }y\}.
  \]

  \vspace{0.2em}
  \textbf{Esempio: distanza di Hamming}
  \[
    d_H(x, y) = \sum_{i=1}^{n} \begin{cases}
                                 0 & \text{se } x_i = y_i \\
                                 1 & \text{se } x_i \neq y_i
    \end{cases}
  \]


\end{frame}

\begin{frame}{Distanza stringa-linguaggio e correction problem}
Per $L\subseteq\Sigma^*$ e $x\in\Sigma^*$:
\[
d(L,x)=\min_{w\in L} d(x,w).
\]

\vspace{0.6em}
\textbf{Correction problem:} trovare una correzione ottima
\[
y=\arg\min_{w\in L} d(x,w).
\]
\end{frame}

\begin{frame}{$k$-vicinato}
Definizione induttiva del vicinato di edit:
\[
E_0(w)=\{w\},\quad
E_{k+1}(w)=E_k(w)\cup\bigcup_{w'\in E_k(w)}\bigcup_{a\in\Sigma}(ins_a(w')\cup del_a(w')).
\]

Per un linguaggio:
\[
E_k(L)=\bigcup_{w\in L}E_k(w).
\]

\begin{itemize}
  \item Risultato: la famiglia dei linguaggi IDPDA e' chiusa rispetto al $k$-vicinato.
\end{itemize}
\end{frame}

\begin{frame}{Risultati noti}
    \begin{itemize}
        \item $O(n)$ per linguaggi regolari.
        \item $O(n^3)$ per linguaggi context-free.
        \item La classe dei linguaggi riconosciuti da automi 1-NAuxPDA è la più ampia per cui si conosce un algoritmo efficiente per il calcolo della distanza di edit.
    \end{itemize}
\end{frame}

\begin{frame}{Classi di complessita' considerate}
\begin{itemize}
  \item $P$: tempo polinomiale deterministico.
  \item $NP$: verifica polinomiale di certificati.
  \item $PSPACE$: spazio polinomiale.
\end{itemize}
\vspace{0.5em}
\begin{columns}[T,onlytextwidth]
    \column{0.5\textwidth}
    Relazioni note:
    \[
    P\subseteq NP\subseteq PSPACE.
    \]
    \column{0.5\textwidth}
    Questione aperta:
    \[
    P \stackrel{?}{=} NP.
    \]
\end{columns}
\end{frame}

\begin{frame}{Collegamento con $P=NP$}
\begin{block}{Teorema}
Se $P=NP$, allora la distanza di edit di qualunque linguaggio in $NP$ e' calcolabile in tempo polinomiale.
\end{block}

Idea della prova:
\begin{itemize}
  \item Si considera il problema decisionale $\Pi_L$: ``$d(L,x)\le k$?''.
  \item Si indovina una stringa candidata $y$ di lunghezza polinomiale in $|x|$.
  \item Si verifica se $y \in L$ e $d(y, x) \le k$.
  \item Si calcola $d(L,x)$ valutando $\Pi_L$ per valori crescenti di $k$.
\end{itemize}
\end{frame}
